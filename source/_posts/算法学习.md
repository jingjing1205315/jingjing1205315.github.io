### 两数之和

```
/**
  * 
  * @param numbers int整型一维数组 
  * @param target int整型 
  * @return int整型一维数组
  */
function twoSum( numbers ,  target ) {
    // write code here
    if(!numbers || numbers.length < 2 ) return [];
    let obj = {[numbers[0]]: 1};
    for(let i = 1,l = numbers.length; i<l; i++){    
        let tem = target - numbers[i];
        if(obj[tem] >= 1){
            return [obj[tem], i+1]
        }
        obj[numbers[i]] = i+1;
    }
}
module.exports = {
    twoSum : twoSum
};
```
